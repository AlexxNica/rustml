pub fn group<T: PartialEq + Clone>(v: &Vec<T>) -> Vec<(T, usize)>

Counts and compresses consecutive elements that are equal.

Example

use rustml::vectors::*;

let a = vec![1, 1, 2, 3, 3, 3, 3, 5, 5, 3];
assert_eq!(
    group(&a),
    vec![(1, 2), (2, 1), (3, 4), (5, 2), (3, 1)]
);
